1. Input Validation

func validateInput(value string, base int) error {
    if strings.TrimSpace(value) == "" {
        return fmt.Errorf("input cannot be empty")
    }
    
    // Remove common prefixes (0x, 0b, 0o)
    cleanValue := removePrefix(value)
    
    // Validate each character against the base
    for _, char := range strings.ToUpper(cleanValue) {
        digitValue := getDigitValue(char)
        if digitValue >= base {
            return fmt.Errorf("invalid digit '%c' for base %d", char, base)
        }
    }
    return nil
}

2. Number System Conversion

func convertFromDecimal(decimal int64, toBase int) string {
    if decimal == 0 {
        return "0"
    }
    
    digits := "0123456789ABCDEF"
    result := ""
    
    for decimal > 0 {
        remainder := decimal % int64(toBase)
        result = string(digits[remainder]) + result
        decimal = decimal / int64(toBase)
    }
    
    return result
}

ğŸ”„ Key Frontend Functions

1. Real-time Conversion Handler

const performConversions = useCallback(async (value: string, fromBase: number, toBase: number) => {
    if (!value.trim()) {
        setMainConversion(null)
        setAutoConversions(null)
        return
    }

    setIsLoading(true)
    try {
        // Get main conversion with steps
        const mainResult = await ConvertNumber(value, fromBase, toBase)
        setMainConversion(mainResult)

        // Get all conversions for auto-conversion card
        const allResult = await ConvertToAllSystems(value, fromBase)
        setAutoConversions(allResult)
    } catch (error) {
        // Error handling...
    } finally {
        setIsLoading(false)
    }
}, [])

2. Debounced Input Processing

useEffect(() => {
    const fromBase = getCurrentSystemInfo(fromSystem)?.base || 10
    const toBase = getCurrentSystemInfo(toSystem)?.base || 2
    
    if (inputValue.trim()) {
        const timeoutId = setTimeout(() => {
            performConversions(inputValue, fromBase, toBase)
        }, 300) // Debounce for 300ms
        
        return () => clearTimeout(timeoutId)
    } else {
        setMainConversion(null)
        setAutoConversions(null)
    }
}, [inputValue, fromSystem, toSystem, performConversions])

3. Portal-based Dropdown System

function PortalDropdown({ isOpen, onClose, buttonRef, children }) {
    const [position, setPosition] = useState({ top: 0, left: 0, width: 0 })

    useEffect(() => {
        if (isOpen && buttonRef.current) {
            const rect = buttonRef.current.getBoundingClientRect()
            setPosition({
                top: rect.bottom + 4,
                left: rect.left,
                width: rect.width
            })
        }
    }, [isOpen, buttonRef])

    return createPortal(
        <div className="fixed z-[99999] ..." style={position}>
            {children}
        </div>,
        document.body
    )
}

            // Initialize rows if needed

âš™ï¸ 9. Technical Challenges Solved

ğŸ”§ Challenge 1: Dropdown Z-Index Issues

Problem: Dropdowns were appearing behind glassmorphic cards due to stacking context issues.

Solution: Implemented React Portals to render dropdowns at document.body level with dynamic positioning.

ğŸ”§ Challenge 2: Text Overflow in Cards

Problem: Long conversion results were breaking card layouts.

Solution: Added break-all, overflow-wrap-anywhere, and custom scrollbars.

ğŸ”§ Challenge 3: Animation Reset on Input

Problem: Background animation was resetting every time user typed.

Solution: Used useRef to maintain persistent animation state independent of parent re-renders.

ğŸ”§ Challenge 4: Type Safety with Wails Bindings

Problem: Ensuring type safety between Go backend and TypeScript frontend.

Solution: Used Wails automatic TypeScript binding generation and custom type definitions.

ğŸ† Performance Optimizations

  • Debounced Input: 300ms delay prevents excessive API calls during typing
  • Memoized Callbacks: useCallback prevents unnecessary re-renders
  • Conditional Rendering: Only render complex components when data is available
  • Efficient State Management: Minimized state updates and effects
